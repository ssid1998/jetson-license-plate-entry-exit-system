706 inline void PROFILER_BEGIN( profilerQuery query )
707 {
708 const uint32_t evt = query*2;
709 const uint32_t flag = (1 << query);
710
711 CUDA(cudaEventRecord(mEventsGPU[evt], mStream));
712 timestamp(&mEventsCPU[evt]);
713
714 mProfilerQueriesUsed |= flag;
715 mProfilerQueriesDone &= ~flag;
716 }
717
721 inline void PROFILER_END( profilerQuery query )
722 {
723 const uint32_t evt = query*2+1;
724
725 CUDA(cudaEventRecord(mEventsGPU[evt]));
726 timestamp(&mEventsCPU[evt]);
727 timespec cpuTime;
728 timeDiff(mEventsCPU[evt-1], mEventsCPU[evt], &cpuTime);
729 mProfilerTimes[query].x = timeFloat(cpuTime);
730
731 if( mEnableProfiler && query == PROFILER_NETWORK )
732 {
733 LogVerbose(LOG_TRT "layer network time - %f ms\n", gProfiler.timingAccumulator);
734 gProfiler.timingAccumulator = 0.0f;
735 LogWarning(LOG_TRT "note -- when processing a single image, run 'sudo jetson_clocks' before\n"
736 " to disable DVFS for more accurate profiling/timing measurements\n");
737 }
738 }
739
743 inline bool PROFILER_QUERY( profilerQuery query )
744 {
745 const uint32_t flag = (1 << query);
746
747 if( query == PROFILER_TOTAL )
748 {
749 mProfilerTimes[PROFILER_TOTAL].x = 0.0f;
750 mProfilerTimes[PROFILER_TOTAL].y = 0.0f;
751
752 for( uint32_t n=0; n < PROFILER_TOTAL; n++ )
753 {
754 if( PROFILER_QUERY((profilerQuery)n) )
755 {
756 mProfilerTimes[PROFILER_TOTAL].x += mProfilerTimes[n].x;
757 mProfilerTimes[PROFILER_TOTAL].y += mProfilerTimes[n].y;
758 }
759 }
760
761 return true;
762 }
763 else if( mProfilerQueriesUsed & flag )
764 {
765 if( !(mProfilerQueriesDone & flag) )
766 {
767 const uint32_t evt = query*2;
768 float cuda_time = 0.0f;
769 CUDA(cudaEventElapsedTime(&cuda_time, mEventsGPU[evt], mEventsGPU[evt+1]));
770 mProfilerTimes[query].y = cuda_time;
771 mProfilerQueriesDone |= flag;
772 //mProfilerQueriesUsed &= ~flag;
773 }
774
775 return true;
776 }
777
778 return false;
779 }
