77inline bool Event::Wait()
78{
79 mQueryMutex.Lock();
80
81 while(!mQuery)

84 if( mAutoReset )

92// Wait
93inline bool Event::Wait( const timespec& timeout )
94{
95 mQueryMutex.Lock();
96
97 const timespec abs_time = timeAdd( timestamp(), timeout );
98
99 while(!mQuery)
100 {
101 const int ret = pthread_cond_timedwait(&mID, mQueryMutex.GetID(), &abs_time);
102
103 if( ret == ETIMEDOUT )
104 {
105 mQueryMutex.Unlock();
106 return false;
107 }
108 }
109
110 if( mAutoReset )
111 mQuery = false;
112
113 mQueryMutex.Unlock();
114 return true;
115}
