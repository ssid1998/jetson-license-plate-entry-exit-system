85inline cudaError_t cudaCheckError(cudaError_t retval, const char* txt, const char* file, int line )
86{
87 if( retval == cudaSuccess )
88 {
89 #if !defined(CUDA_TRACE)
90 return cudaSuccess;
91 #else
92 LogDebug(LOG_CUDA "%s\n", txt);
93 #endif
94 }
95 else
96 {
97 LogError(LOG_CUDA "%s\n", txt);
98 LogError(LOG_CUDA " %s (error %u) (hex 0x%02X)\n", cudaGetErrorString(retval), retval, retval);
99 LogError(LOG_CUDA " %s:%i\n", file, line);
100 }
101
102 return retval;
103}
