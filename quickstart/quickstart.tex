	\section{Hardware Commissioning}
	\textbf{Prerequisite:} Ensure the Main Unit (Jetson) is placed on a flat, non-conductive surface.
	
	% itemsep=2pt adds small gaps between steps
	\begin{enumerate}[leftmargin=*, itemsep=2pt, topsep=2pt] 
		\item \textbf{Connect Peripherals (Before Powering On):}
		\begin{itemize}[noitemsep, topsep=0pt] % Keep sub-items tight
			\item \textbf{Monitor:} Connect the HDMI cable to the HDMI port.
			\item \textbf{Camera:} Connect to a \textbf{Blue (USB 3.0)} port for high-speed data.
			\item \textbf{Controls:} Connect the Mouse/Keyboard to a \textbf{Black (USB 2.0)} port.
		\end{itemize}
		
		\item \textbf{Connect Power:}
		\begin{itemize}[noitemsep, topsep=0pt]
			\item Insert the USB-C connector into the port labeled \textbf{"DC IN"}.
			\item Plug the adapter into a wall outlet. The \textbf{Green LED} on the chassis will light up.
		\end{itemize}
	\end{enumerate}
	
	\section{Software Initialization}
	The system boots to a command line. Follow these steps to start the Operator Dashboard:
	
	\begin{enumerate}[leftmargin=*, itemsep=2pt, topsep=2pt]
		\item \textbf{System Login:}
		\begin{itemize}[noitemsep, topsep=0pt]
			\item When prompted for \texttt{recomputer login:}, type \textbf{\texttt{nvidia}} and press Enter.
			\item When prompted for \texttt{Password:}, type \textbf{\texttt{nvidia}} and press Enter.
			\item \textit{(Note: Password characters will not be visible on screen)}
		\end{itemize}
		
		\item \textbf{Launch Interface:}
		\begin{itemize}[noitemsep, topsep=0pt]
			\item Type the command: \textbf{\texttt{startx}}
			\item Press \textbf{Enter}. The graphical interface will load automatically.
		\end{itemize}
	\end{enumerate}
	
	\section{Operation \& Status}
	\begin{itemize}[leftmargin=*, itemsep=2pt]
		\item \textbf{Live View:} The left panel shows the real-time camera feed.
		\item \textbf{Logging:} The right panel lists detected plates.
	\end{itemize}
	
	\section{Safe Shutdown (CRITICAL)}
	\textbf{WARNING:} Never unplug the power cable while the system is running.
	
	\begin{enumerate}[leftmargin=*, itemsep=2pt, topsep=2pt]
		\item Locate the \textbf{Red "SHUTDOWN"} button on the screen.
		\item Click it once. Wait for the monitor to go black.
		\item Only then disconnect the power adapter.
	\end{enumerate}
