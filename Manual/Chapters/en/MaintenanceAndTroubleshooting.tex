\chapter{Maintenance and Troubleshooting}

\section{Updating JetPack SDK}
\begin{itemize}[label=--,left=1.5em]
	\item Regularly check for JetPack SDK updates at \href{https://developer.nvidia.com/embedded/jetpack}{NVIDIA JetPack SDK page}.
	\item To update, run:
	\begin{verbatim}
		sudo apt-get update
		sudo apt-get upgrade
	\end{verbatim}
\end{itemize}

\subsection{Device Not Booting}
\begin{itemize}
	\item Ensure the power supply is adequate (5V 4A recommended).
	\item Reflash the MicroSD card using the latest JetPack version.
\end{itemize}

\subsection{Overheating}
If the device overheats:
\begin{itemize}
	\item Ensure proper ventilation.
	\item Add a fan or heatsink for additional cooling.
\end{itemize}
